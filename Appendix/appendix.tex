\chapter{Joint Power/Torque/Speed Calculations}
\label{apx:JointPowerTorqueSpeedCalcs}

\begin{equation}
    P_{output} = \epsilon P_{input} 
     \label{eq:PowerCalc}
\end{equation}

\begin{equation}
    \tau_{output} =   \frac{n_1}{n_2}  \epsilon  \tau_{input}
    \label{eq:TorqueCalc}
\end{equation}

\begin{equation}
    \omega_{output} = \omega_{input}  \frac{n_2}{n_1}
    \label{eq:SpeedCalc}
\end{equation}
